컨테이너 조회

# sudo docker ps -a

우분투 최신 이미지 가져오기

# sudo docker pull ubuntu:latest

컨테이너 생성

# sudo docker run -it --name nodejs ubuntu:14.04 /bin/bash

컨테이너 생성(포트연결)

# sudo docker run -it -p 3000:3000 --name nodejs ubuntu:14.04 /bin/bash

# sudo docker run -it -p 3000-3010:3000-3010/tcp --name nodejs ubuntu:14.04 /bin/bash

# sudo docker run -it -p 3000-3020:3000-3020/tcp --name nodejs --link mysql nodejs-dinga.kr /bin/bash

컨테이너 시작

# sudo docker start ubuntu:14.04

컨테이너 이름 변경

# sudo docker rename hello nodejs

시작된 컨테이너에 접속

# sudo docker attach nodejs

컨테이너 탈출

# ctrl + p, ctrl + q

외부에서 컨테이너의 명령을 실행

# sudo docker exec nodejs date

컨테이너 정지

# sudo docker stop nodejs

컨테이너 삭제

# sudo docker rm nodejs

이미지 삭제

# sudo docker rmi ubuntu:latest
